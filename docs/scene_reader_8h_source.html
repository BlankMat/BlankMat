52 {
53 // Process each vertex
54 for (unsigned int i = 0; i < mesh->mNumVertices; i++)

68 // Processes the indices of the mesh
69 static void ProcessAssimpIndices(const std::vector<Vertex>& vertices, std::vector<unsigned int>& outIndices, const aiMesh* assimpMesh)
70 {
71 // Process each index
72 for (unsigned int i = 0; i < assimpMesh->mNumFaces; i++)
73 {
74 if (assimpMesh->mFaces[i].mNumIndices > 0 && vertices[assimpMesh->mFaces[i].mIndices[0]].tangent == glm::vec3(0.0f))
75 {
76 // If vertex does not have tangents, calculate them
77 Vertex vertex1 = vertices[assimpMesh->mFaces[i].mIndices[0]];
78 Vertex vertex2 = vertices[assimpMesh->mFaces[i].mIndices[1]];
79 Vertex vertex3 = vertices[assimpMesh->mFaces[i].mIndices[2]];
80 glm::vec3 edge1 = vertex2.pos - vertex1.pos;
81 glm::vec3 edge2 = vertex3.pos - vertex1.pos;
82 glm::vec2 deltaUV1 = vertex2.texCoords - vertex1.texCoords;
83 glm::vec2 deltaUV2 = vertex3.texCoords - vertex1.texCoords;
84 glm::vec3 tangent = glm::vec3(0.0f);
85
86 float f = 1.0f / (deltaUV1.x * deltaUV2.y - deltaUV2.x * deltaUV1.y);
87 tangent.x = f * (deltaUV2.y * edge1.x - deltaUV1.y * edge2.x);
88 tangent.y = f * (deltaUV2.y * edge1.y - deltaUV1.y * edge2.y);
89 tangent.z = f * (deltaUV2.y * edge1.z - deltaUV1.y * edge2.z);
90
91 vertex1.tangent = tangent;
92 vertex2.tangent = tangent;
93 vertex3.tangent = tangent;
94 }
95 for (unsigned int j = 0; j < assimpMesh->mFaces[i].mNumIndices; j++)
96 {
97 outIndices.push_back(assimpMesh->mFaces[i].mIndices[j]);
98 }
99 }

103 static Material* ProcessAssimpMaterial(Scene* scene, const aiMesh* assimpMesh, const aiScene* assimpScene)
104 {
105 // If there are no textures, use default material
106 if (assimpMesh->mMaterialIndex >= assimpScene->mNumMaterials)

109 // Process each material's diffuse and specular maps
110 aiMaterial* material = assimpScene->mMaterials[assimpMesh->mMaterialIndex];
111
112 // If the material already exists, return it
113 Material* newMaterial = scene->GetMaterial(material->GetName().C_Str());
114 if (newMaterial != nullptr)
115 return newMaterial;
116
117 // 1. diffuse maps
118 std::vector<Texture*> diffuseMaps = LoadMaterialTextures(scene, material, aiTextureType_DIFFUSE, TextureType::DIFFUSE);
119 if (diffuseMaps.empty())
120 diffuseMaps.push_back(scene->GetTexture("default_diffuse"));
121 // 2. ambient maps
122 std::vector<Texture*> ambientMaps = LoadMaterialTextures(scene, material, aiTextureType_AMBIENT, TextureType::AMBIENT);
123 if (ambientMaps.empty())
124 {
125 if (!diffuseMaps.empty())
126 {
127 Texture* ambientTexture = new Texture(diffuseMaps[0]->id, TextureType::AMBIENT, diffuseMaps[0]->path, diffuseMaps[0]->name);
128 scene->AddTexture(ambientTexture->name, ambientTexture);
129 ambientMaps.push_back(ambientTexture);
130 }
131 else
132 {
133 ambientMaps.push_back(scene->GetTexture("default_ambient"));
134 }
135 }
136 // 3. specular maps
137 std::vector<Texture*> specularMaps = LoadMaterialTextures(scene, material, aiTextureType_SPECULAR, TextureType::SPECULAR);
138 if (specularMaps.empty())
139 specularMaps.push_back(scene->GetTexture("default_specular"));
140 // 4. normal maps
141 std::vector<Texture*> normalMaps = LoadMaterialTextures(scene, material, aiTextureType_HEIGHT, TextureType::NORMAL);
142 if (normalMaps.empty())
143 normalMaps.push_back(scene->GetTexture("default_normal"));
144 // 5. height maps
145 std::vector<Texture*> heightMaps = LoadMaterialTextures(scene, material, aiTextureType_DISPLACEMENT, TextureType::HEIGHT);
146 if (heightMaps.empty())
147 heightMaps.push_back(scene->GetTexture("default_height"));
148 // 6. alpha maps
149 std::vector<Texture*> alphaMaps = LoadMaterialTextures(scene, material, aiTextureType_OPACITY, TextureType::ALPHA);
150 if (alphaMaps.empty())
151 alphaMaps.push_back(scene->GetTexture("default_alpha"));
152
153 // Read properties into material
154 aiColor3D diffuse;
155 material->Get(AI_MATKEY_COLOR_DIFFUSE, diffuse);
156 glm::vec3 kd = Vec3FromAssimp(diffuse);
157
158 aiColor3D ambient;
159 material->Get(AI_MATKEY_COLOR_AMBIENT, ambient);
160 glm::vec3 ka = Vec3FromAssimp(ambient);
161

164 glm::vec3 ks = Vec3FromAssimp(specular);
165
166 aiColor3D emissive;

169
170 float ns;
171 material->Get(AI_MATKEY_SHININESS, ns);
172
173 float ni;
174 material->Get(AI_MATKEY_REFRACTI, ni);
175
176 float d;
177 material->Get(AI_MATKEY_OPACITY, d);
178
179 // Make material out of loaded textures
180 newMaterial = new Material(material->GetName().C_Str(), diffuseMaps, ambientMaps, specularMaps, normalMaps, heightMaps, alphaMaps,
181 ka, kd, ks, ns, ni, d, ke);
182 scene->AddMaterial(material->GetName().C_Str(), newMaterial);

236 size_t lastSlash = path.find_last_of('/') + 1;
