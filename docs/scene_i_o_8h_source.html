268 bool PromptSaveChanges(GLFWwindow* window, State* state)
269 {
270 // If state is saved, break
271 if (state->IsSaved())
272 return true;
273
274 // If state is not saved, set a popup
275 pfd::button res = pfd::message(
276 "Unsaved Changes",
277 "You have made unsaved changes to the scene. Would you like to save them before opening a new scene?",
278 pfd::choice::yes_no_cancel,
279 pfd::icon::warning
280 ).result();
281
282 switch (res)
283 {
284 case pfd::button::yes:
285 // If the user requested to save first, run the save command
286 SaveScene();
287 return true;
288 break;
289 case pfd::button::no:
290 // If the user did not request to save, don't run the save command
291 return true;
292 break;
293 case pfd::button::cancel:
294 default:
295 // If the user canceled, do not continue
296 return false;
297 }
298 }
