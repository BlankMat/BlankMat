255 bool PromptSaveChanges(GLFWwindow* window, State* state)
256 {
257 // If state is saved, break
258 if (state->IsSaved())
259 return true;
260
261 // If state is not saved, set a popup
262 pfd::button res = pfd::message(

265 pfd::choice::yes_no_cancel,
266 pfd::icon::warning
267 ).result();
268
269 switch (res)

271 case pfd::button::yes:
272 // If the user requested to save first, run the save command
273 SaveScene();

277 // If the user did not request to save, don't run the save command
