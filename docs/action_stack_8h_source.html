107 void AppendNode(ICommand* command)
108 {
109 // Don't add empty or untrackable commands
110 if (command == nullptr || !command->Trackable())
111 return;
112
113 // Attempt to combine the new command with the previous one. If successful, don't change anything.
114 if (mCurNode != nullptr && mCurNode->Combine(command))
115 return;
116
117 // If there is no undo, clear all previous commands
118 if (!command->Undoable())
119 Clear();
120

122 auto* newNode = new ActionNode(mCurNode, nullptr, command);
123 mCurSize++;
124
125 // If no nodes have been added, make this the first and only node
126 if (mCurNode == nullptr || mLastNode == nullptr)

128 mCurNode = newNode;
129 mLastNode = newNode;
130 mFirstNode->mNextNode = newNode;
131 return;
132 }
133
134 // If an interaction happens after an undo, remove redo branch
135 if (mCurNode->HasNext())

137 ClearAfter(mCurNode);
138 mLastNode = mCurNode;
139 }
140
141 // If there is at least one node, move the last and current node pointers
142 mLastNode->mNextNode = newNode;
143 newNode->mPrevNode = mLastNode;
