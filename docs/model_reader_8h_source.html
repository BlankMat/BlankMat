28 static Mesh* ProcessMesh(IScene* scene, aiMesh* mesh, const aiScene* assimpScene)
29 {
30 std::vector<Vertex> vertices;
31 std::vector<unsigned int> indices;
32 Material* newMaterial;
33
34 // Process each vertex
35 for (unsigned int i = 0; i < mesh->mNumVertices; i++)
36 {
37 glm::vec3 pos = Vec3FromAssimp(mesh->mVertices[i]);
38 glm::vec3 normal = Vec3FromAssimp(mesh->mNormals[i]);
39 glm::vec2 texCoords = (mesh->mTextureCoords[0]) ? Vec2FromAssimp(mesh->mTextureCoords[0][i]) : glm::vec2();
40 glm::vec3 tangent = glm::vec3(0.0f);
41 if (mesh->mTangents != nullptr && mesh->mTangents->Length() > 0)
42 {
43 tangent = Vec3FromAssimp(mesh->mTangents[i]);
44 }
45 vertices.push_back(Vertex(pos, normal, texCoords, tangent));
46 }
47
48 // Process each index
49 for (unsigned int i = 0; i < mesh->mNumFaces; i++)
50 {
51 if (mesh->mFaces[i].mNumIndices > 0 && vertices[mesh->mFaces[i].mIndices[0]].tangent == glm::vec3(0.0f))
52 {
53 // If vertex does not have tangents, calculate them
54 Vertex vertex1 = vertices[mesh->mFaces[i].mIndices[0]];
55 Vertex vertex2 = vertices[mesh->mFaces[i].mIndices[1]];
56 Vertex vertex3 = vertices[mesh->mFaces[i].mIndices[2]];
57 glm::vec3 edge1 = vertex2.pos - vertex1.pos;
58 glm::vec3 edge2 = vertex3.pos - vertex1.pos;
59 glm::vec2 deltaUV1 = vertex2.texCoords - vertex1.texCoords;
60 glm::vec2 deltaUV2 = vertex3.texCoords - vertex1.texCoords;
61 glm::vec3 tangent = glm::vec3(0.0f);
62
63 float f = 1.0f / (deltaUV1.x * deltaUV2.y - deltaUV2.x * deltaUV1.y);
64 tangent.x = f * (deltaUV2.y * edge1.x - deltaUV1.y * edge2.x);
65 tangent.y = f * (deltaUV2.y * edge1.y - deltaUV1.y * edge2.y);
66 tangent.z = f * (deltaUV2.y * edge1.z - deltaUV1.y * edge2.z);
67
68 vertex1.tangent = tangent;
69 vertex2.tangent = tangent;
70 vertex3.tangent = tangent;
71 }
72 for (unsigned int j = 0; j < mesh->mFaces[i].mNumIndices; j++)
73 {
74 indices.push_back(mesh->mFaces[i].mIndices[j]);
75 }
76 }
77
78 // Process each material's diffuse and specular maps
79 if (mesh->mMaterialIndex >= 0)
80 {
81 aiMaterial* material = assimpScene->mMaterials[mesh->mMaterialIndex];
82 newMaterial = scene->GetMaterial(material->GetName().C_Str());
83 if (newMaterial == nullptr)
84 {
85 // 1. diffuse maps
86 std::vector<Texture*> diffuseMaps = LoadMaterialTextures(scene, material, aiTextureType_DIFFUSE, "texture_diffuse");
87 if (diffuseMaps.size() == 0)
88 diffuseMaps.push_back(scene->GetTexture("default_diffuse"));
89 // 2. ambient maps
90 std::vector<Texture*> ambientMaps = LoadMaterialTextures(scene, material, aiTextureType_AMBIENT, "texture_ambient");
91 if (ambientMaps.size() == 0)
92 {
93 if (diffuseMaps.size() > 0)
94 {
95 Texture* ambientTexture = new Texture(diffuseMaps[0]->id, "texture_ambient", diffuseMaps[0]->path, diffuseMaps[0]->name);
96 scene->AddTexture(ambientTexture->name, ambientTexture);
97 ambientMaps.push_back(ambientTexture);
98 }
99 else
100 {
101 ambientMaps.push_back(scene->GetTexture("default_ambient"));
102 }
103 }
104 // 3. specular maps
105 std::vector<Texture*> specularMaps = LoadMaterialTextures(scene, material, aiTextureType_SPECULAR, "texture_specular");
106 if (specularMaps.size() == 0)
107 specularMaps.push_back(scene->GetTexture("default_specular"));
108 // 4. normal maps
109 std::vector<Texture*> normalMaps = LoadMaterialTextures(scene, material, aiTextureType_HEIGHT, "texture_normal");
110 if (normalMaps.size() == 0)
111 normalMaps.push_back(scene->GetTexture("default_normal"));
112 // 5. height maps
113 std::vector<Texture*> heightMaps = LoadMaterialTextures(scene, material, aiTextureType_DISPLACEMENT, "texture_height");
114 if (heightMaps.size() == 0)
115 heightMaps.push_back(scene->GetTexture("default_height"));
116 // 6. alpha maps
117 std::vector<Texture*> alphaMaps = LoadMaterialTextures(scene, material, aiTextureType_OPACITY, "texture_alpha");
118 if (alphaMaps.size() == 0)
119 alphaMaps.push_back(scene->GetTexture("default_alpha"));
120
121 // Read properties into material
122 aiColor3D diffuse;
123 material->Get(AI_MATKEY_COLOR_DIFFUSE, diffuse);
124 glm::vec3 kd = Vec3FromAssimp(diffuse);
125
126 aiColor3D ambient;
127 material->Get(AI_MATKEY_COLOR_AMBIENT, ambient);
128 glm::vec3 ka = Vec3FromAssimp(ambient);
129

132 glm::vec3 ks = Vec3FromAssimp(specular);
133
134 aiColor3D emissive;
135 material->Get(AI_MATKEY_COLOR_EMISSIVE, emissive);
136 glm::vec3 ke = Vec3FromAssimp(emissive);
137
138 float ns;
139 material->Get(AI_MATKEY_SHININESS, ns);
140
141 float ni;
142 material->Get(AI_MATKEY_REFRACTI, ni);
143
144 float d;
145 material->Get(AI_MATKEY_OPACITY, d);
146
147 // Make material out of loaded textures
148 newMaterial = new Material(material->GetName().C_Str(), diffuseMaps, ambientMaps, specularMaps, normalMaps, heightMaps, alphaMaps,
149 ka, kd, ks, ns, ni, d, ke);
150 scene->AddMaterial(material->GetName().C_Str(), newMaterial);
151 }
152 }
153 // If there are no textures, use default material
154 else
155 {
156 newMaterial = scene->GetMaterial("default");
157 }
158
159 return new Mesh(mesh->mName.C_Str(), vertices, indices, newMaterial);

163 static std::vector<Texture*> LoadMaterialTextures(IScene* scene, aiMaterial* mat, aiTextureType type, const std::string& typeName)
164 {
165 std::vector<Texture*> textures;
166 for (unsigned int i = 0; i < mat->GetTextureCount(type); i++)
167 {
168 aiString str;
169 mat->GetTexture(type, i, &str);
170 std::string path = str.C_Str();
171 std::string name = path.substr(0, path.find_last_of('.'));
172 std::string textureName = scene->GetName() + "_" + name;
173
174 // Check if texture is loaded already
175 Texture* texture = scene->GetTexture(textureName);
176 if (texture != nullptr)
177 {
178 textures.push_back(texture);
179 continue;
180 }
181
182 // if texture hasn't been loaded already, load it
183 texture = new Texture(typeName, scene->GetDirectory(), path, textureName);
184 textures.push_back(texture);
185 scene->AddTexture(textureName, texture);
186 std::cout << " - Loaded texture " << textureName << std::endl;
187 }
188 return textures;
