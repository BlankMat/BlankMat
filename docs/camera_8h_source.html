64 // Calculates the basis of the camera
65 void CalcBasis()
66 {
67 mDir = glm::normalize(glm::vec3(
68 cos(mRot.y) * sin(mRot.x),
69 sin(mRot.y),
70 cos(mRot.y) * cos(mRot.x)
71 ));
72

76 // Up vector : perpendicular to both direction and right
77 mUp = glm::normalize(glm::cross(mRight, mDir));
