38 // Calculates the basis of the camera
39 void CalcBasis()
40 {
41 glm::vec3 rot = mRot.Get();
42 mDir.SetNoUpdate(glm::normalize(glm::vec3(
43 cos(rot.y) * sin(rot.x),
44 sin(rot.y),
45 cos(rot.y) * cos(rot.x)
46 )));
47
48 // Right vector
49 mRight.SetNoUpdate(glm::normalize(glm::cross((glm::vec3)mDir, glm::vec3(0.0f, 1.0f, 0.0f))));
50
51 // Up vector : perpendicular to both direction and right
52 mUp.SetNoUpdate(glm::normalize(glm::cross((glm::vec3)mRight, (glm::vec3)mDir)));
53 mTarget.SetNoUpdate(mPos + mDir);

55 mRecalcRotation = true;

144 // Returns the projection matrix of the camera
145 const glm::mat4& GetProjection(float aspect)
146 {
147 // Only recalculate projection matrix if needed
148 if (aspect != 0.0f && (mRecalcProjection || aspect != mPrevAspect))
149 {
150 // Projection
151 glm::mat4 projection = glm::mat4(1.0f);
152
153 // Perspective projection
154 if (mIsPerspective)
155 {
156 projection = glm::perspective(glm::radians(+mFOV), aspect, +mNearClip, +mFarClip);
157 }
158 // Orthographic projection
159 else
160 {
161 projection = glm::ortho(-mOrthSize * aspect, +mOrthSize * aspect, -mOrthSize, +mOrthSize, +mNearClip, +mFarClip);
162 }
163 mProjection = projection;
164 mRecalcProjection = false;
165 mPrevAspect = aspect;
166 }
