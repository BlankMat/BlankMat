27static glm::vec3 GetRotationDegrees(const glm::vec3& dir, const glm::vec3& up = glm::vec3(0,1,0))
28{
29 //return glm::vec3(atan2(dir.x, dir.z), asin(dir.y), 0.0f);
30 /*
31 According to "Calculation of Euler Angles" by Prokopi Nikolaev,
32 variables defined here use y as the up (Z1 in the formula)
33 dir = x -> X1
34 up = y -> Z1
35 right = z -> Y1
36 */
37
38 /*
39 glm::vec3 right = glm::normalize(glm::cross(dir, up));
40 float y1xz = glm::sqrt(up.x * up.x + up.z * up.z);
41 if (y1xz > 0.0f)
42 {
43 return glm::vec3(
44 glm::degrees(atan2f(right.x * up.z - right.z * up.x, dir.x * up.z - dir.z * up.x)),
45 glm::degrees(atan2f(y1xz, up.y)),
46 glm::degrees(-atan2f(-up.x, up.z))
47 );
48 }
49 else
50 {
51 return glm::vec3(
52 glm::degrees(0.0f),
53 glm::degrees((up.y > 0.0f ? 0.0f : PI)),
54 glm::degrees(-atan2f(dir.z, dir.x))
55 );
56 }
57 */
58
59 glm::vec3 right = glm::normalize(glm::cross(dir, up));
60 float rotX = asinf(-dir.y);
61 if (cosf(rotX) > 0.0001)
62 return glm::degrees(glm::vec3(rotX, atan2f(-dir.x, -dir.z), atan2f(right.y, up.y)));
63 return glm::degrees(glm::vec3(rotX, 0.0f, atan2f(-up.x, right.x)));
64}

76static const glm::vec4 RotateAround(const glm::vec4& aPointToRotate, const glm::vec4& aRotationCenter, const glm::mat4& aRotationMatrix)
77{
78 glm::mat4 translate = glm::translate(glm::mat4(1.0f), glm::vec3(aRotationCenter.x, aRotationCenter.y, aRotationCenter.z));
79 glm::mat4 invTranslate = glm::inverse(translate);
80
81 // The idea:
82 // 1) Translate the object to the center
83 // 2) Make the rotation
84 // 3) Translate the object back to its original location
85 glm::mat4 transform = translate * aRotationMatrix * invTranslate;
86
87 return transform * aPointToRotate;
88}

147static const void ParseStringByDelim(std::vector<std::string>& out, const std::string& str, const std::string& delim)
148{
149 // Make sure out vec is cleared
150 out.clear();
151
152 // Parse each part of the string
153 size_t spos = 0;
154 std::string token;
155 std::string tempStr = TrimWhitespace(str);
156 while ((spos = tempStr.find(delim)) != std::string::npos) {
157 // Parse token
158 token = tempStr.substr(0, spos);
159 out.push_back(token);
160 tempStr.erase(0, spos + delim.length());
161 }
162 out.push_back(tempStr);
163}

218static std::string IncrementName(const std::string& name, int padLen = 3)
219{
220 size_t numIndex = name.find_last_not_of("0123456789");
221 std::string endNums = name.substr(numIndex + 1);
222
223 // If the name does not end in numbers, pad it with zeros
224 if (endNums.length() <= 0)
225 return name + "_" + GetPadding(padLen, "0");
226
227 // If the name ends in numbers, increment the number by one
228 return name.substr(0, numIndex + 1) + IntToString(std::stoi(endNums) + 1, endNums.length());
229}

300static const std::string DecimalToHex(int value)
301{
302 // char array to store hexadecimal number
303 char hexValue[2] = {};
304
305 // counter for hexadecimal number array
306 int i = 0;
307 while (value != 0)
308 {
309 // Store remainder
310 int temp = value % 16;
311 value = value / 16;
312
313 hexValue[i] = temp + (temp < 10 ? 48 : 55);
314 i++;
315 }
316
317 std::string hexCode = "";
318 if (i == 2)
319 {
320 hexCode.push_back(hexValue[0]);
321 hexCode.push_back(hexValue[1]);
322 }
323 else if (i == 1)
324 {
325 hexCode = "0";
326 hexCode.push_back(hexValue[0]);
327 }
328 else if (i == 0)
329 {
330 hexCode = "00";
331 }
332
333 // Return the equivalent hexadecimal color code
334 return hexCode;
335}

342static const std::string Vec3ToHex(const glm::vec3& vec)
343{
344 int r = Clamp((int)round(vec.r * 255), 0, 255);
345 int g = Clamp((int)round(vec.g * 255), 0, 255);

349 hex += DecimalToHex(r);
350 hex += DecimalToHex(g);
351 hex += DecimalToHex(b);
