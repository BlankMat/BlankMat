27static glm::vec3 GetRotationDegrees(const glm::vec3& dir, const glm::vec3& up = glm::vec3(0,1,0))
28{
29 //return glm::vec3(atan2(dir.x, dir.z), asin(dir.y), 0.0f);
30 /*
31 According to "Calculation of Euler Angles" by Prokopi Nikolaev,
32 variables defined here use y as the up (Z1 in the formula)
33 dir = x -> X1
34 up = y -> Z1
35 right = z -> Y1
36 */
37
38 /*
39 glm::vec3 right = glm::normalize(glm::cross(dir, up));
40 float y1xz = glm::sqrt(up.x * up.x + up.z * up.z);
41 if (y1xz > 0.0f)
42 {
43 return glm::vec3(
44 glm::degrees(atan2f(right.x * up.z - right.z * up.x, dir.x * up.z - dir.z * up.x)),
45 glm::degrees(atan2f(y1xz, up.y)),
46 glm::degrees(-atan2f(-up.x, up.z))
47 );
48 }
49 else
50 {
51 return glm::vec3(
52 glm::degrees(0.0f),
53 glm::degrees((up.y > 0.0f ? 0.0f : PI)),
54 glm::degrees(-atan2f(dir.z, dir.x))
55 );
56 }
57 */
58
59 glm::vec3 right = glm::normalize(glm::cross(dir, up));
60 float rotX = asinf(-dir.y);
61 if (cosf(rotX) > 0.0001)
62 return glm::degrees(glm::vec3(rotX, atan2f(-dir.x, -dir.z), atan2f(right.y, up.y)));
63 return glm::degrees(glm::vec3(rotX, 0.0f, atan2f(-up.x, right.x)));
64}

76static const glm::vec4 RotateAround(const glm::vec4& aPointToRotate, const glm::vec4& aRotationCenter, const glm::mat4& aRotationMatrix)
77{
78 glm::mat4 translate = glm::translate(glm::mat4(1.0f), glm::vec3(aRotationCenter.x, aRotationCenter.y, aRotationCenter.z));
79 glm::mat4 invTranslate = glm::inverse(translate);
80
81 // The idea:
82 // 1) Translate the object to the center
83 // 2) Make the rotation
84 // 3) Translate the object back to its original location
85 glm::mat4 transform = translate * aRotationMatrix * invTranslate;
86
87 return transform * aPointToRotate;
88}

239static const void ParseStringByDelim(std::vector<std::string>& out, const std::string& str, const std::string& delim)
240{
241 // Make sure out vec is cleared
242 out.clear();
243
244 // Parse each part of the string
245 size_t spos = 0;
246 std::string token;
247 std::string tempStr = TrimWhitespace(str);
248 while ((spos = tempStr.find(delim)) != std::string::npos) {
249 // Parse token
250 token = tempStr.substr(0, spos);
251 out.push_back(token);
252 tempStr.erase(0, spos + delim.length());
253 }
254 out.push_back(tempStr);
255}

326static std::string IncrementName(const std::string& name, int padLen = 3)
327{
328 size_t numIndex = name.find_last_not_of("0123456789");
329 std::string endNums = name.substr(numIndex + 1);
330
331 // If the name does not end in numbers, pad it with zeros
332 if (endNums.length() <= 0)
333 return name + "_" + GetPadding(padLen, "0");
334
335 // If the name ends in numbers, increment the number by one
336 return name.substr(0, numIndex + 1) + IntToString(std::stoi(endNums) + 1, endNums.length());
337}

408static const std::string DecimalToHex(int value)
409{
410 // char array to store hexadecimal number
411 char hexValue[2] = {};
412
413 // counter for hexadecimal number array
414 int i = 0;
415 while (value != 0)
416 {
417 // Store remainder
418 int temp = value % 16;
419 value = value / 16;
420
421 hexValue[i] = temp + (temp < 10 ? 48 : 55);
422 i++;
423 }
424
425 std::string hexCode = "";
426 if (i == 2)
427 {
428 hexCode.push_back(hexValue[0]);
429 hexCode.push_back(hexValue[1]);
430 }
431 else if (i == 1)
432 {
433 hexCode = "0";
434 hexCode.push_back(hexValue[0]);
435 }
436 else if (i == 0)
437 {
438 hexCode = "00";
439 }
440
441 // Return the equivalent hexadecimal color code
442 return hexCode;
443}

450static const std::string Vec3ToHex(const glm::vec3& vec)
451{
452 int r = Clamp((int)round(vec.r * 255), 0, 255);
453 int g = Clamp((int)round(vec.g * 255), 0, 255);
454 int b = Clamp((int)round(vec.b * 255), 0, 255);
455
456 std::string hex = "#";
457 hex += DecimalToHex(r);
458 hex += DecimalToHex(g);
459 hex += DecimalToHex(b);
460 return hex;
461}
