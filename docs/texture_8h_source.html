111 static unsigned int TextureFromColor(const glm::vec3& color, const std::string& name)
112 {
113 unsigned int textureID;
114 glGenTextures(1, &textureID);
115
116 unsigned char data[3] = { 0, 0, 0 };
117 Vec3ToRGB(color, data);
118
119 glBindTexture(GL_TEXTURE_2D, textureID);
120 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, 1, 1, 0, GL_RGB, GL_UNSIGNED_BYTE, data);
121 glGenerateMipmap(GL_TEXTURE_2D);
122
123 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
124 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
125 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
126 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
127
128 return textureID;

138 static unsigned int TextureFromFile(const std::string& directory, const std::string& name, bool gamma = false)
139 {
140 std::string fileName = directory + '/' + name;
141 unsigned int textureID;
142 glGenTextures(1, &textureID);
143
144 int width;
145 int height;
146 int nrComponents;
147 unsigned char* data = stbi_load(fileName.c_str(), &width, &height, &nrComponents, 0);
148 if (data)
149 {
150 GLenum format = GL_RGB;
151 if (nrComponents == 1)
152 format = GL_RED;
153 else if (nrComponents == 3)
154 format = GL_RGB;
155 else if (nrComponents == 4)
156 format = GL_RGBA;
157
158 glBindTexture(GL_TEXTURE_2D, textureID);
159 glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
160 glGenerateMipmap(GL_TEXTURE_2D);
161
162 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
163 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
164 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
165 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
166
167 stbi_image_free(data);
168 }
169 else
170 {
171 std::cout << "Texture failed to load at path: " << name << std::endl;
172 stbi_image_free(data);
173 }
174 return textureID;
175 }
