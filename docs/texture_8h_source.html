157 static unsigned int TextureFromFile(const std::string& directory, const std::string& filename, int& outWidth, int& outHeight, bool flip = true)
158 {
159 stbi_set_flip_vertically_on_load(flip);
160 std::string fileName = directory + '/' + filename;
161 unsigned int textureID;
162 glGenTextures(1, &textureID);
163 int numComponents;
164 unsigned char* data = stbi_load(fileName.c_str(), &outWidth, &outHeight, &numComponents, 0);
165 if (data)
166 {
167 GLenum format = GL_RGB;
168 if (numComponents == 1)
169 format = GL_RED;
170 else if (numComponents == 3)
171 format = GL_RGB;
172 else if (numComponents == 4)
173 format = GL_RGBA;
174
175 glBindTexture(GL_TEXTURE_2D, textureID);
176 glTexImage2D(GL_TEXTURE_2D, 0, format, outWidth, outHeight, 0, format, GL_UNSIGNED_BYTE, data);
177 glGenerateMipmap(GL_TEXTURE_2D);
178
179 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
180 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
181 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
182 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
183
184 stbi_image_free(data);
185 }
186 else
187 {
188 std::cout << "Texture failed to load at path: " << fileName << std::endl;
189 stbi_image_free(data);
190 }
191 return textureID;
192 }

201 static void CopyTexture(const std::string& file, const std::string& fromDir, const std::string& toDir, bool flip = true)
202 {
203 stbi_set_flip_vertically_on_load(flip);
204 stbi_flip_vertically_on_write(flip);
205 int width;
206 int height;
207 std::string startFile = fromDir + '/' + file;
208 std::string endFile = toDir + '/' + file;
209 int numComponents;

211 if (data)
212 {
213 std::string ext = GetFileExtension(file);
214 if (ext == ".jpg")
215 {
216 stbi_write_jpg(endFile.c_str(), width, height, numComponents, data, width * numComponents);
217 }
218 else if (ext == ".png")
219 {

221 }
222 else if (ext == ".tga")
223 {
224 stbi_write_tga(endFile.c_str(), width, height, numComponents, data);
225 }
226 stbi_image_free(data);
227 }
